316 // Standard permits specialization of std::numeric_limits. This specialization
317 // is used to resolve ambiguity between isinf and std::isinf in glibc:
318 // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=48891
319 // and the same for isnan and signbit.
320 template <>
321 class numeric_limits<fmt::internal::dummy_int> :
322  public std::numeric_limits<int> {
323  public:
324  // Portable version of isinf.
325  template <typename T>
326  static bool isinfinity(T x) {
327  using namespace fmt::internal;
328  // The resolution "priority" is:
329  // isinf macro > std::isinf > ::isinf > fmt::internal::isinf
330  if (const_check(sizeof(isinf(x)) != sizeof(dummy_int)))
331  return isinf(x) != 0;
332  return !_finite(static_cast<double>(x));
333  }
334 
335  // Portable version of isnan.
336  template <typename T>
337  static bool isnotanumber(T x) {
338  using namespace fmt::internal;
339  if (const_check(sizeof(isnan(x)) != sizeof(fmt::internal::dummy_int)))
340  return isnan(x) != 0;
341  return _isnan(static_cast<double>(x)) != 0;
342  }
343 
344  // Portable version of signbit.
345  static bool isnegative(double x) {
346  using namespace fmt::internal;
347  if (const_check(sizeof(signbit(x)) != sizeof(fmt::internal::dummy_int)))
348  return signbit(x) != 0;
349  if (x < 0) return true;
350  if (!isnotanumber(x)) return false;
351  int dec = 0, sign = 0;
352  char buffer[2]; // The buffer size must be >= 2 or _ecvt_s will fail.

509 template <typename T, std::size_t SIZE = inline_buffer_size,
510  typename Allocator = std::allocator<T> >
511 class basic_memory_buffer: private Allocator, public internal::basic_buffer<T> {
512  private:
513  T store_[SIZE];
514 
515  // Deallocate memory allocated by the buffer.
516  void deallocate() {
517  T* data = this->data();
518  if (data != store_) Allocator::deallocate(data, this->capacity());
519  }
520 
521  protected:
522  void grow(std::size_t size) FMT_OVERRIDE;
523 
524  public:
525  explicit basic_memory_buffer(const Allocator &alloc = Allocator())
526  : Allocator(alloc) {
527  this->set(store_, SIZE);
528  }
529  ~basic_memory_buffer() { deallocate(); }
530 
531  private:
532  // Move data from other to this buffer.
533  void move(basic_memory_buffer &other) {
534  Allocator &this_alloc = *this, &other_alloc = other;
535  this_alloc = std::move(other_alloc);
536  T* data = other.data();
537  std::size_t size = other.size(), capacity = other.capacity();
538  if (data == other.store_) {
539  this->set(store_, capacity);
540  std::uninitialized_copy(other.store_, other.store_ + size,
541  internal::make_checked(store_, capacity));
542  } else {
543  this->set(data, capacity);
544  // Set pointer to the inline array so that delete is not called
545  // when deallocating.
546  other.set(other.store_, 0);
547  }
548  this->resize(size);
549  }
550 
551  public:
558  basic_memory_buffer(basic_memory_buffer &&other) {
559  move(other);
560  }
561 
567  basic_memory_buffer &operator=(basic_memory_buffer &&other) {
568  assert(this != &other);
569  deallocate();
570  move(other);
571  return *this;
572  }
573 
574  // Returns a copy of the allocator associated with this buffer.
575  Allocator get_allocator() const { return *this; }
576 };

578 template <typename T, std::size_t SIZE, typename Allocator>
579 void basic_memory_buffer<T, SIZE, Allocator>::grow(std::size_t size) {
580  std::size_t old_capacity = this->capacity();
581  std::size_t new_capacity = old_capacity + old_capacity / 2;
582  if (size > new_capacity)
583  new_capacity = size;
584  T *old_data = this->data();
585  T *new_data = internal::allocate<Allocator>(*this, new_capacity);
586  // The following code doesn't throw, so the raw pointer above doesn't leak.
587  std::uninitialized_copy(old_data, old_data + this->size(),
588  internal::make_checked(new_data, new_capacity));
589  this->set(new_data, new_capacity);
590  // deallocate must not throw according to the standard, but even if it does,
591  // the buffer already uses the new storage and will deallocate it in
592  // destructor.
593  if (old_data != store_)
594  Allocator::deallocate(old_data, old_capacity);
595 }

955 // An lg handler that formats a decimal number.
956 // Usage: lg(n, decimal_formatter(buffer));
957 class decimal_formatter {
958  private:
959  char *buffer_;
960 
961  void write_pair(unsigned N, uint32_t index) {
962  std::memcpy(buffer_ + N, data::DIGITS + index * 2, 2);
963  }
964 
965  public:
966  explicit decimal_formatter(char *buf) : buffer_(buf) {}
967 
968  template <unsigned N> char *on(uint32_t u) {
969  if (N == 0) {
970  *buffer_ = static_cast<char>(u) + '0';
971  } else if (N == 1) {
972  write_pair(0, u);
973  } else {
974  // The idea of using 4.32 fixed-point numbers is based on
975  // https://github.com/jeaiii/itoa
976  unsigned n = N - 1;
977  unsigned a = n / 5 * n * 53 / 16;
978  uint64_t t = ((1ULL << (32 + a)) /
979  data::ZERO_OR_POWERS_OF_10_32[n] + 1 - n / 9);
980  t = ((t * u) >> a) + n / 5 * 4;
981  write_pair(0, t >> 32);
982  for (unsigned i = 2; i < N; i += 2) {
983  t = 100ULL * static_cast<uint32_t>(t);
984  write_pair(i, t >> 32);
985  }
986  if (N % 2 == 0) {
987  buffer_[N] = static_cast<char>(
988  (10ULL * static_cast<uint32_t>(t)) >> 32) + '0';
989  }
990  }
991  return buffer_ += N + 1;
992  }

1051 // thousands_sep is a functor that is called after writing each char to
1052 // add a thousands separator if necessary.
1053 template <typename UInt, typename Char, typename ThousandsSep>
1054 inline Char *format_decimal(Char *buffer, UInt value, unsigned num_digits,
1055  ThousandsSep thousands_sep) {
1056  buffer += num_digits;
1057  Char *end = buffer;
1058  while (value >= 100) {
1059  // Integer division is slow so do it for a group of two digits instead
1060  // of for every digit. The idea comes from the talk by Alexandrescu
1061  // "Three Optimization Tips for C++". See speed-test for a comparison.
1062  unsigned index = static_cast<unsigned>((value % 100) * 2);
1063  value /= 100;
1064  *--buffer = data::DIGITS[index + 1];
1065  thousands_sep(buffer);
1066  *--buffer = data::DIGITS[index];
1067  thousands_sep(buffer);
1068  }
1069  if (value < 10) {
1070  *--buffer = static_cast<char>('0' + value);
1071  return end;
1072  }
1073  unsigned index = static_cast<unsigned>(value * 2);
1074  *--buffer = data::DIGITS[index + 1];
1075  thousands_sep(buffer);
1076  *--buffer = data::DIGITS[index];
1077  return end;

1919 FMT_CONSTEXPR Iterator parse_arg_id(Iterator it, IDHandler &&handler) {
1920  typedef typename std::iterator_traits<Iterator>::value_type char_type;
1921  char_type c = *it;
1922  if (c == '}' || c == ':') {
1923  handler();
1924  return it;
1925  }
1926  if (c >= '0' && c <= '9') {
1927  unsigned index = parse_nonnegative_int(it, handler);
1928  if (*it != '}' && *it != ':') {
1929  handler.on_error("invalid format string");
1930  return it;

1932  handler(index);
1933  return it;
1934  }

1936  handler.on_error("invalid format string");
1937  return it;
1938  }
1939  auto start = it;
1940  do {
1941  c = *++it;
1942  } while (is_name_start(c) || ('0' <= c && c <= '9'));
1943  handler(basic_string_view<char_type>(
1944  pointer_from(start), to_unsigned(it - start)));
