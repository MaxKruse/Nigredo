58 FMT_FUNC void vprint_rgb(rgb fd, string_view format, format_args args)
59 {
60  char escape_fd[] = "\x1b[38;2;000;000;000m";
61  static FMT_CONSTEXPR_DECL const char RESET_COLOR[] = "\x1b[0m";
62  internal::to_esc(fd.r, escape_fd, 7);
63  internal::to_esc(fd.g, escape_fd, 11);
64  internal::to_esc(fd.b, escape_fd, 15);
65 
66  std::fputs(escape_fd, stdout);
67  vprint(format, args);
68  std::fputs(RESET_COLOR, stdout);
69 }

71 FMT_FUNC void vprint_rgb(rgb fd, rgb bg, string_view format, format_args args)
72 {
73  char escape_fd[] = "\x1b[38;2;000;000;000m"; // foreground color
74  char escape_bg[] = "\x1b[48;2;000;000;000m"; // background color
75  static FMT_CONSTEXPR_DECL const char RESET_COLOR[] = "\x1b[0m";
76  internal::to_esc(fd.r, escape_fd, 7);
77  internal::to_esc(fd.g, escape_fd, 11);
78  internal::to_esc(fd.b, escape_fd, 15);
79 
80  internal::to_esc(bg.r, escape_bg, 7);
81  internal::to_esc(bg.g, escape_bg, 11);
82  internal::to_esc(bg.b, escape_bg, 15);
83 
84  std::fputs(escape_fd, stdout);
85  std::fputs(escape_bg, stdout);
86  vprint(format, args);
87  std::fputs(RESET_COLOR, stdout);
88 }
