144 #include "spdlog/fmt/bin_to_hex.h"
145 void binary_example()
146 {
147  std::vector<char> buf;
148  for (int i = 0; i < 80; i++)
149  {
150  buf.push_back(static_cast<char>(i & 0xff));
151  }
152  spdlog::info("Binary example: {}", spdlog::to_hex(buf));
153  spdlog::info("Another binary example:{:n}", spdlog::to_hex(std::begin(buf), std::begin(buf) + 10));
154  // more examples:
155  // logger->info("uppercase: {:X}", spdlog::to_hex(buf));
156  // logger->info("uppercase, no delimiters: {:Xs}", spdlog::to_hex(buf));
157  // logger->info("uppercase, no delimiters, no position info: {:Xsp}", spdlog::to_hex(buf));
158 }
